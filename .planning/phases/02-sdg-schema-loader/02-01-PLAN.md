pub struct ServiceDefinition {
    pub schema_version: String,

    pub model: ModelDefinition,

    pub api: ApiConfig,

    pub name: String,
    #[serde(default)]
    pub description: String,
    #[serde(default = "default_owner")]

    pub aggregates: HashMap<String, Aggregate>,
}

pub struct Aggregate {
    pub fields: HashMap<String, FieldDefinition>,
    pub states: Vec<String>,

    pub initial_state: Option<String>,
    #[serde(default)]
    pub transitions: HashMap<String, Transition>,
}

pub struct FieldDefinition {
    #[serde(rename = "type")]
    pub field_type: String,  // "string", "integer", "uuid", "uuid[]", etc. per D-09
    #[serde(default)]
    pub required: bool,
    #[serde(default)]
    pub default: Option<serde_json::Value>,
    // Validation properties per D-10
    #[serde(default)]
    pub min: Option<serde_json::Number>,
    #[serde(default)]
    pub max: Option<serde_json::Number>,
    #[serde(default)]
    pub min_length: Option<u64>,
    #[serde(default)]
    pub max_length: Option<u64>,
    #[serde(default)]
    pub pattern: Option<String>,
    #[serde(default)]
    pub format: Option<String>,
    #[serde(default)]  // D-11
    pub references: Option<String>,
    #[serde(default)]
    pub description: Option<String>,

// D-07: from accepts string or array
#[derive(Debug, Clone, Serialize, Deserialize)]
#[serde(untagged)]
pub enum StateRef {
    Single(String),
    Multiple(Vec<String>),
}

pub struct Transition {
    pub from: StateRef,
    pub to: String,  // D-06: "$same" sentinel allowed
    #[serde(default)]
    pub command: Option<CommandDefinition>,
    #[serde(default)]  // D-21: references computation node ID
    pub guard: Option<String>,
    #[serde(default)]  // D-22: auto_fields map
    pub auto_fields: HashMap<String, String>,
    #[serde(default)]  // D-23: override event name
    pub event_name: Option<String>,
    #[serde(default)]
    pub description: Option<String>,
}

pub struct CommandDefinition {
    #[serde(default)]
    pub fields: HashMap<String, FieldDefinition>,
}

// Computations section per D-12
pub struct ComputationsDefinition {
    #[serde(default)]
    pub nodes: Vec<ComputationNode>,
    #[serde(default)]
    pub edges: Vec<Edge>,
}

// D-13: Each node has id, type, params
pub struct ComputationNode {
    pub id: String,
    #[serde(rename = "type")]
    pub node_type: String,  // from function catalog: field, command, context, literal, lookup, etc.
    #[serde(default)]
    pub params: serde_json::Map<String, serde_json::Value>,
}

// D-14: Each edge has from, to, port, optional index
pub struct Edge {
    pub from: String,
    pub to: String,
    pub port: String,
    #[serde(default)]
    pub index: Option<u32>,
}

// API section per D-38 (parsed into types, semantic validation deferred to Phase 6)
pub struct ApiConfig {
    #[serde(default = "default_expose")]
    pub expose: String,  // "all", "none", "explicit"
    #[serde(default = "default_base_path")]
    pub base_path: String,
    #[serde(default = "default_protocol")]
    pub protocol: String,  // "http", "grpc", "both"
    #[serde(default)]
    pub overrides: HashMap<String, serde_json::Value>,
    #[serde(default)]
    pub custom_queries: HashMap<String, CustomQuery>,
}

pub struct CustomQuery {
    pub source: String,
    #[serde(default)]
    pub filter_by: Option<String>,
    #[serde(default)]
    pub fields: Vec<String>,

    pub group_by: Option<String>,
    #[serde(default)]
    pub aggregation: Option<serde_json::Value>,

Implement `Default` for `ComputationsDefinition` (empty nodes and edges) and `ApiConfig` (expose: "all", base_path: "/api", protocol: "http"). Use `StateRef` to handle D-07 (from accepts string or array).

For D-36 (literal node output_type): The `output_type` param is inside the `params` map on `ComputationNode`. The JSON Schema enforces it via `if/then` conditional. Type-level validation of literal nodes happens in semantic pass (Plan 03).

For D-15 (leaf node types): The node types (field, command, context, literal) are string values in the `node_type` field. The JSON Schema enumerates them. The Rust types do NOT use a tagged enum for nodes because the v2 format has a flat `type` field with many possible values from the function catalog. Instead use a generic `ComputationNode` struct with `node_type: String` and `params: Map`. This is the correct approach for SDG v2 where the function catalog is extensible. Type-specific validation happens in the semantic pass.

**Step 3: Create `crates/sdg-loader/src/error.rs`** with the complete error hierarchy.

```rust
use thiserror::Error;
use std::path::PathBuf;

#[derive(Debug, Error)]
pub enum SdgError {
    // File system errors
    #[error("Failed to read SDG file '{path}': {source}")]
    FileRead {
        path: PathBuf,
        source: std::io::Error,
    },

    // JSON parse errors
    #[error("Invalid JSON in SDG file: {message}")]
    JsonParse { message: String },

    // Pass 1: Schema violations
    #[error("[schema] {instance_path}: {message}")]
    SchemaViolation {
        instance_path: String,
        schema_path: String,
        message: String,
    },

    // Pass 2: Version errors
    #[error("Missing 'schema_version' field in SDG")]
    MissingVersion,

    #[error("Invalid schema version '{value}': {reason}")]
    InvalidVersion { value: String, reason: String },

    #[error("Incompatible schema version {found} (runtime supports major version {expected_major})")]
    IncompatibleVersion {
        found: String,
        expected_major: u64,
    },

    // Deserialization errors (between pass 2 and 3)
    #[error("Failed to deserialize SDG into typed structures: {message}")]
    Deserialization { message: String },

    // Pass 3: Semantic errors
    #[error("[semantic] {path}: state '{name}' not found in aggregate '{aggregate}'{suggestion}")]
    InvalidStateReference {
        path: String,
        name: String,
        aggregate: String,
        suggestion: String,
    },

    #[error("[semantic] {path}: field '{name}' not found in aggregate '{aggregate}'{suggestion}")]
    InvalidFieldReference {
        path: String,
        name: String,
        aggregate: String,
        suggestion: String,
    },

    #[error("[semantic] {path}: type mismatch on edge - expected {expected}, found {found}")]
    TypeMismatch {
        path: String,
        expected: String,
        found: String,
    },

    #[error("[semantic] {path}: {message}")]
    SemanticError { path: String, message: String },

    #[error("[semantic] {path}: unknown node type '{node_type}'{suggestion}")]
    UnknownNodeType {
        path: String,
        node_type: String,
        suggestion: String,
    },

    #[error("[semantic] {path}: duplicate node ID '{node_id}'")]
    DuplicateNodeId { path: String, node_id: String },

    #[error("[semantic] {path}: unknown context path '{context_path}'{suggestion}")]
    InvalidContextPath {
        path: String,
        context_path: String,
        suggestion: String,
    },

    #[error("[semantic] {path}: implicit field name '{name}' cannot be declared by user")]
    ImplicitFieldConflict { path: String, name: String },

    // Pass 4: DAG errors
    #[error("[dag] Cycle detected involving node '{node}' in computation graph")]
    DagCycle { node: String },

    #[error("[dag] Edge references non-existent node: {from} -> {to}")]
    DagEdgeReference { from: String, to: String },

    #[error("[dag] Edge targets non-existent port '{port}' on node '{node}' (type: {node_type})")]
    DagInvalidPort {
        port: String,
        node: String,
        node_type: String,
    },
}

impl SdgError {
    /// Returns which validation pass caught this error
    pub fn pass(&self) -> &'static str {
        match self {
            Self::FileRead { .. } | Self::JsonParse { .. } => "file",
            Self::SchemaViolation { .. } => "schema",
            Self::MissingVersion | Self::InvalidVersion { .. } | Self::IncompatibleVersion { .. } => "version",
            Self::Deserialization { .. } => "deserialization",
            Self::InvalidStateReference { .. }
            | Self::InvalidFieldReference { .. }
            | Self::TypeMismatch { .. }
            | Self::SemanticError { .. }
            | Self::UnknownNodeType { .. }
            | Self::DuplicateNodeId { .. }
            | Self::InvalidContextPath { .. }
            | Self::ImplicitFieldConflict { .. } => "semantic",
            Self::DagCycle { .. }
            | Self::DagEdgeReference { .. }
            | Self::DagInvalidPort { .. } => "dag",
        }
    }
}
```

Note: Use `Box` to wrap large variants if clippy::large_enum_variant fires. The `source` field on `FileRead` contains `std::io::Error` which is not Clone -- make `SdgError` derive `Debug` only (not `Clone`). Add `PartialEq` where needed for testing by comparing string representations.

**Step 5: Create `crates/sdg-loader/src/schema.rs`** with the embedded schema and validator factory.

```rust
use jsonschema::Validator;

pub const SDG_SCHEMA_STR: &str = include_str!("sdg_schema.json");

/// Creates a JSON Schema validator from the embedded SDG schema.
/// Panics if the embedded schema is invalid (programming error, not user error).
pub fn create_validator() -> Validator {
    let schema: serde_json::Value = serde_json::from_str(SDG_SCHEMA_STR)
        .expect("embedded SDG schema must be valid JSON");
    jsonschema::draft202012::new(&schema)
        .expect("embedded SDG schema must be valid Draft 2020-12")
}
```
