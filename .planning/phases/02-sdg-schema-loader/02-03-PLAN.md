**suggestions.rs** -- Implement the string similarity module per D-09:
```rust
use strsim::normalized_damerau_levenshtein;

/// Find the closest match from candidates using Damerau-Levenshtein distance.
/// Returns a suggestion string like ". Did you mean 'InProgress'?" if a match
/// with > 0.6 similarity is found. Returns None if no good match exists.
pub fn suggest_similar(input: &str, candidates: &[&str]) -> Option<String> {
    candidates
        .iter()
        .map(|c| (*c, normalized_damerau_levenshtein(input, c)))
        .filter(|(_, score)| *score > 0.6)
        .max_by(|(_, a), (_, b)| a.partial_cmp(b).unwrap_or(std::cmp::Ordering::Equal))
        .map(|(candidate, _)| format!(". Did you mean '{candidate}'?"))
}
```

**dag.rs** -- Implement `materialize_dag` function. The `ComputationDag` struct already exists from Plan 02-01. Add the construction function:

```rust
use petgraph::graph::{DiGraph, NodeIndex};
use petgraph::algo::toposort;
use std::collections::HashMap;
use crate::types::{DagNode, GuardDef};
use crate::error::SdgError;

#[derive(Debug, Clone)]
pub struct ComputationDag {
    pub graph: DiGraph<DagNode, ()>,
    pub topo_order: Vec<NodeIndex>,
    pub node_map: HashMap<String, NodeIndex>,
    pub output_node: String,
}

/// Materialize a ComputationDag from a GuardDef.
/// Adds all nodes to a petgraph DiGraph, adds edges, then runs toposort().
/// toposort() returns Err(Cycle) if the graph has cycles, combining cycle detection
/// and topological ordering in a single O(|V|+|E|) pass per research.
///
/// context parameter: identifies which transition this guard belongs to (e.g., "transition 'CompleteTask' guard")
pub fn materialize_dag(guard: &GuardDef, context: &str) -> Result<ComputationDag, Vec<SdgError>> {
    let mut graph = DiGraph::new();
    let mut node_map = HashMap::new();
    let mut errors = Vec::new();

    // Add nodes
    for (id, node) in &guard.nodes {
        let idx = graph.add_node(node.clone());
        node_map.insert(id.clone(), idx);
    }

    // Add edges (validate references)
    for (from, to) in &guard.edges {
        match (node_map.get(from), node_map.get(to)) {
            (Some(&src), Some(&dst)) => {
                graph.add_edge(src, dst, ());
            }
            _ => {
                if node_map.get(from).is_none() {
                    errors.push(SdgError::DagEdgeReference {
                        from: from.clone(),
                        to: to.clone(),
                    });
                }
                if node_map.get(to).is_none() && node_map.get(from).is_some() {
                    errors.push(SdgError::DagEdgeReference {
                        from: from.clone(),
                        to: to.clone(),
                    });
                }
            }
        }
    }

    // Validate output node exists
    if !node_map.contains_key(&guard.output) {
        errors.push(SdgError::DagMissingOutput {
            context: context.to_string(),
            output: guard.output.clone(),
        });
    }

    if !errors.is_empty() {
        return Err(errors);
    }

    // toposort() detects cycles and computes topological order in one pass
    match toposort(&graph, None) {
        Ok(order) => Ok(ComputationDag {
            graph,
            topo_order: order,
            node_map,
            output_node: guard.output.clone(),
        }),
        Err(cycle) => {
            // Find the node ID from the NodeIndex involved in the cycle
            let cycle_node_id = node_map.iter()
                .find(|(_, &idx)| idx == cycle.node_id())
                .map(|(id, _)| id.clone())
                .unwrap_or_else(|| format!("unknown (index {:?})", cycle.node_id()));

            Err(vec![SdgError::DagCycle {
                node: cycle_node_id,
                context: context.to_string(),
            }])
        }
    }
}
```

use crate::types::ServiceDefinition;
use crate::error::SdgError;
use crate::dag;

/// Pass 4: Materialize computation DAGs from all guard definitions.
/// Detects cycles and computes topological order for each DAG.
/// Collects errors across all guards before returning (per D-10).
pub fn run(definition: &ServiceDefinition) -> Result<Vec<(String, dag::ComputationDag)>, Vec<SdgError>> {
    let mut all_errors = Vec::new();
    let mut dags = Vec::new();

    for (transition_name, transition) in &definition.transitions {
        if let Some(guard) = &transition.guard {
            let context = format!("transition '{transition_name}' guard");
            match dag::materialize_dag(guard, &context) {
                Ok(computation_dag) => {
                    dags.push((transition_name.clone(), computation_dag));
                }
                Err(errors) => {
                    all_errors.extend(errors);
                }
            }
        }
    }

    if all_errors.is_empty() {
        Ok(dags)
    } else {
        Err(all_errors)
    }
}
```

pub mod schema_pass;
pub mod version_pass;
pub mod semantic_pass;
pub mod dag_pass;

use crate::error::SdgError;
use crate::types::{ServiceDefinition, ValidatedSdg};

/// Run the full 4-pass validation pipeline per D-11.
/// Returns ValidatedSdg with materialized DAGs on success.
/// Returns all errors from the first failing pass on failure.
pub fn validate(raw: &serde_json::Value) -> Result<ValidatedSdg, Vec<SdgError>> {
    // Pass 1: Schema conformance
    let errors = schema_pass::run(raw);
    if !errors.is_empty() {
        return Err(errors);
    }

    // Pass 2: Version compatibility
    let errors = version_pass::run(raw);
    if !errors.is_empty() {
        return Err(errors);
    }

    // Deserialize into typed structs
    let definition: ServiceDefinition = serde_json::from_value(raw.clone())
        .map_err(|e| vec![SdgError::Deserialization {
            message: e.to_string(),
        }])?;

    // Pass 3: Semantic validation
    let errors = semantic_pass::run(&definition);
    if !errors.is_empty() {
        return Err(errors);
    }

    // Pass 4: DAG materialization + cycle detection
    let dags_result = dag_pass::run(&definition)?;
    let dags = dags_result.into_iter().collect();

    // Collect warnings for deferred sections per D-01
    let mut warnings = Vec::new();
    if definition.external_dependencies.is_some() {
        warnings.push("external_dependencies section present but not processed in this runtime version".to_string());
    }
    if definition.processes.is_some() {
        warnings.push("processes section present but not processed in this runtime version".to_string());
    }

    Ok(ValidatedSdg {
        definition,
        dags,
        warnings,
    })
}
```
