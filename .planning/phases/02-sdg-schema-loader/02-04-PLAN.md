```rust
impl SdgError {
    /// Which validation pass produced this error.
    pub fn pass_name(&self) -> &'static str {
        match self {
            Self::FileRead { .. } | Self::JsonParse { .. } => "parse",
            Self::SchemaViolation { .. } => "schema",
            Self::MissingVersion | Self::InvalidVersion { .. } | Self::IncompatibleVersion { .. } => "version",
            Self::Deserialization { .. } => "deserialization",
            Self::InvalidStateReference { .. } | Self::TypeMismatch { .. }
            | Self::CompletenessError { .. } | Self::DuplicateName { .. } => "semantic",
            Self::DagCycle { .. } | Self::DagEdgeReference { .. } | Self::DagMissingOutput { .. } => "dag",
        }
    }

    /// Format as a human-readable error line per D-09.
    /// Format: "[pass] path: message (suggestion)"
    pub fn format_human(&self) -> String {
        // Use the Display impl from thiserror, which already has the format.
        // But wrap with pass prefix for consistency.
        format!("{self}")
    }

    /// Format as a JSON object for machine-readable output per D-12.
    pub fn format_json(&self) -> serde_json::Value {
        serde_json::json!({
            "pass": self.pass_name(),
            "message": self.to_string(),
            "error_type": self.error_type_name(),
        })
    }

    /// Snake_case name of the error variant for JSON output.
    fn error_type_name(&self) -> &'static str {
        match self {
            Self::FileRead { .. } => "file_read",
            Self::JsonParse { .. } => "json_parse",
            Self::SchemaViolation { .. } => "schema_violation",
            Self::MissingVersion => "missing_version",
            Self::InvalidVersion { .. } => "invalid_version",
            Self::IncompatibleVersion { .. } => "incompatible_version",
            Self::Deserialization { .. } => "deserialization",
            Self::InvalidStateReference { .. } => "invalid_state_reference",
            Self::TypeMismatch { .. } => "type_mismatch",
            Self::CompletenessError { .. } => "completeness_error",
            Self::DuplicateName { .. } => "duplicate_name",
            Self::DagCycle { .. } => "dag_cycle",
            Self::DagEdgeReference { .. } => "dag_edge_reference",
            Self::DagMissingOutput { .. } => "dag_missing_output",

    }
```

**Add standalone formatting functions:**

```rust
/// Format a list of errors as human-readable output per D-12 default format.
/// Each error on its own line, prefixed with a number.
pub fn format_errors_human(errors: &[SdgError]) -> String {
    if errors.is_empty() {
        return String::from("No errors.");
    }
    let mut output = format!("SDG validation failed ({} error{}):\n",
        errors.len(),
        if errors.len() == 1 { "" } else { "s" }
    );
    for (i, error) in errors.iter().enumerate() {
        output.push_str(&format!("  {}. {}\n", i + 1, error.format_human()));
    }
    output
}

/// Format a list of errors as machine-readable JSON per D-12 --json flag.
pub fn format_errors_json(errors: &[SdgError]) -> serde_json::Value {
    serde_json::json!({
        "valid": false,
        "error_count": errors.len(),
        "errors": errors.iter().map(SdgError::format_json).collect::<Vec<_>>(),
    })
}
```
