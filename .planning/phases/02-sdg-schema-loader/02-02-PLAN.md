**version_pass.rs** -- Implement Pass 2 (SemVer version compatibility per D-03, D-11):
```rust
use semver::Version;
use crate::error::SdgError;

/// The major version that this runtime version supports.
const SUPPORTED_MAJOR_VERSION: u64 = 1;

/// Pass 2: Check schema_version field for SemVer compatibility.
/// Major version must match SUPPORTED_MAJOR_VERSION. Minor/patch differences accepted per D-03.
pub fn run(raw: &serde_json::Value) -> Vec<SdgError> {
    let version_str = match raw.get("schema_version").and_then(|v| v.as_str()) {
        Some(v) => v,
        None => return vec![SdgError::MissingVersion],
    };

    let version = match Version::parse(version_str) {
        Ok(v) => v,
        Err(e) => return vec![SdgError::InvalidVersion {
            value: version_str.to_string(),
            reason: e.to_string(),
        }],
    };

    if version.major != SUPPORTED_MAJOR_VERSION {
        return vec![SdgError::IncompatibleVersion {
            found: version,
            expected_major: SUPPORTED_MAJOR_VERSION,
        }];
    }

    vec![]
}
```

```rust
    // Pass 1: Schema conformance

            message: e.to_string(),

    Ok(definition)
}
```

```rust
use std::path::Path;

pub mod types;
pub mod error;
pub mod schema;
pub mod dag;
pub mod suggestions;
pub mod validation;

pub use types::ServiceDefinition;
pub use error::SdgError;

/// Load and validate an SDG file from the given path.
///
/// Runs the multi-pass validation pipeline per D-10/D-11:
/// 1. Read file from disk
/// 2. Parse as JSON
/// 3. Pass 1: JSON Schema conformance
/// 4. Pass 2: Version compatibility
/// 5. Deserialize into typed structs
/// 6. Pass 3: Semantic validation (added in Plan 02-04)
/// 7. Pass 4: DAG materialization (added in Plan 02-04)
///
/// Returns the fully validated ServiceDefinition or all errors from the first failing pass.
pub fn load(path: &Path) -> Result<ServiceDefinition, Vec<SdgError>> {
    // Read file
    let content = std::fs::read_to_string(path)
        .map_err(|e| vec![SdgError::FileRead {
            path: path.to_owned(),
            source: e,
        }])?;

    // Parse JSON
    let raw: serde_json::Value = serde_json::from_str(&content)
        .map_err(|e| vec![SdgError::JsonParse { source: e }])?;

    // Run validation pipeline (passes 1-2 + deserialization)
    validation::validate(&raw)
}
```
